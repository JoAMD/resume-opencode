   ```typescript
   export function unifiedDiffText(a: string, b: string, labelA: string, labelB: string): string {
     const result = structuredPatch(labelA, labelB, a, b, '', '');
     if (!result.hunks || result.hunks.length === 0) {
       return '';
     }
   
     const header = `--- ${labelA}\n+++ ${labelB}\n`;
     const hunks = result.hunks.map(hunk => {
       const hunkLines = hunk.lines.map(line => {
         // structuredPatch prefixes each line with ' ' (context), '-' (removed), or '+' (added)
         // We keep the prefix and format the line after it
         const prefix = line[0];
         const content = line.slice(1);
         if (prefix === '-') return `-${content}`;
         if (prefix === '+') return `+${content}`;
         return ` ${content}`;
       }).join('\n');
       return `@@ -${hunk.oldStart},${hunk.oldLines} +${hunk.newStart},${hunk.newLines} @@\n${hunkLines}`;
     }).join('\n');
   
     return header + hunks + '\n';
   }
   ```

   function wrapDiffLinesWithSpans(diffText) {
     const lines = diffText.split('\n');
     return lines.map(line => {
       if (line.startsWith('-')) {
         // Removed line: escape content (not the - prefix), then wrap
         return '<span class="diff-removed">- ' + escapeHtml(line.slice(1)) + '</span>';
       }
       if (line.startsWith('+')) {
         // Added line: escape content (not the + prefix), then wrap
         return '<span class="diff-added">+ ' + escapeHtml(line.slice(1)) + '</span>';
       }
       // Context lines (space-prefixed), hunk headers (---, +++, @@), etc. — pass through as plain text
       return escapeHtml(line);
     }).join('\n');
   }
