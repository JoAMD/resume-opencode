Replace the existing `wrapDiffLinesWithSpans` function (lines 296-312 of public/suggestions.js) with the correct implementation below.

The new function must:
- Scan lines sequentially
- When it encounters a hunk header line matching the regex `/^@@ -(\d+)(?:,\d+)? \+(\d+)(?:,\d+)? @@/`:
  - Extract oldStart (captured group 1) and newStart (captured group 2) from the header
  - Set oldLine = oldStart and newLine = newStart for the next content line
  - Output the header line unchanged (no line number span)
- For every other line:
  - If line starts with `-`: use and increment oldLine (then oldLine++)
  - If line starts with `+`: use and increment newLine (then newLine++)
  - Otherwise (context line): use and increment both (oldLine++, newLine++)
  - Include the appropriate line-number span (diff-line-num--changed for +/- lines)
- Do NOT use array index (idx) for line numbering

Exact replacement:

```js
  function wrapDiffLinesWithSpans(diffText) {
    const lines = diffText.split('\n');
    let oldLine = 1;
    let newLine = 1;
    return lines.map((line) => {
      const hunkMatch = line.match(/^@@ -(\d+)(?:,\d+)? \+(\d+)(?:,\d+)? @@/);
      if (hunkMatch) {
        oldLine = parseInt(hunkMatch[1], 10);
        newLine = parseInt(hunkMatch[2], 10);
        return escapeHtml(line);
      }
      const isRemoved = line.startsWith('-');
      const isAdded = line.startsWith('+');
      const displayOld = isAdded ? '' : String(oldLine);
      const displayNew = isAdded ? '' : String(newLine);
      if (isRemoved) oldLine++;
      else if (isAdded) newLine++;
      else { oldLine++; newLine++; }
      if (isRemoved) {
        return `<span class="diff-line-num">${displayOld}</span><span class="diff-line-num diff-line-num--changed">${displayNew}</span><span class="diff-removed">- ${escapeHtml(line.slice(1))}</span>`;
      }
      if (isAdded) {
        return `<span class="diff-line-num">${displayOld}</span><span class="diff-line-num diff-line-num--changed">${displayNew}</span><span class="diff-added">+ ${escapeHtml(line.slice(1))}</span>`;
      }
      return `<span class="diff-line-num">${displayOld}</span><span class="diff-line-num">${displayNew}</span>${escapeHtml(line)}`;
    }).join('\n');
  }
```
